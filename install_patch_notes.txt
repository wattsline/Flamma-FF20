# Add this near the end of install.sh, after smoke_test:

create_app_launcher() {
  if [[ "$(uname -s)" != "Darwin" ]]; then
    return
  fi

  echo
  read -r -p "Create FF20 Native.app in ~/Applications? [Y/n] " answer
  answer="${answer:-Y}"

  if [[ "$answer" =~ ^[Yy]$ ]]; then
    if [[ -x "./create_macos_app.sh" ]]; then
      ./create_macos_app.sh
    else
      warn "create_macos_app.sh not found; skipping app launcher."
    fi
  else
    warn "Skipping macOS app launcher."
  fi
}

# Then call it before the final Installation complete section:
create_app_launcher
